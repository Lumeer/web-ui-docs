import {NgZone} from '@angular/core';

interface MovePosition {
  fromIndex: number;
  toIndex: number;
}

export class ColumnLayout {
  private muuri: any;
  private movePosition: MovePosition;

  public constructor(selector: string, options: any, zone: NgZone, onMove?: (event: MovePosition) => any) {
    setTimeout(() => {
      if (!ColumnLayout.containerElementExists(selector)) {
        return;
      }

      zone.runOutsideAngular(() => {
        this.muuri = new window['Muuri'](selector, options);
        this.muuri.on('move', (event: MovePosition) => {
          if (this.movePosition) {
            this.movePosition.toIndex = event.toIndex;
          } else {
            this.movePosition = event;
          }
        });
        this.muuri.on('dragReleaseEnd', () => {
          if (this.movePosition) {
            onMove(this.movePosition);
            this.movePosition = null;
          }
        });
      });
    });
  }

  public destroy() {
    if (this.muuri) {
      this.muuri.destroy();
      this.muuri = null;
    }
  }

  public refreshItems() {
    if (this.muuri) {
      this.muuri.refreshItems();
    }
  }

  public layout() {
    if (this.muuri) {
      this.muuri.layout();
    }
  }

  private static containerElementExists(selector: string): boolean {
    return !!document.querySelector(selector);
  }
}
